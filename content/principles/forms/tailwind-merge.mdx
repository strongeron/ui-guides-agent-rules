<Callout type="quote" title="Tailwind Golden Rule">
  Use tailwind-merge when building reusable form components that accept className props. It intelligently resolves conflicting utilities.
</Callout>

## Why This Matters

Form components often have base styles that consumers want to override. Without tailwind-merge, adding `className="p-6"` to a component with internal `p-4` results in both classes existing - with unpredictable results. tailwind-merge detects conflicts and keeps only the last value.

## Implementation

<CodeBlock title="Form Component" language="tsx">
{`import { twMerge } from 'tailwind-merge';

interface InputProps extends React.InputHTMLAttributes<HTMLInputElement> {
  className?: string;
}

export function Input({ className, ...props }: InputProps) {
  return (
    <input
      className={twMerge(
        'w-full px-3 py-2 border rounded-md',
        'focus:outline-none focus:ring-2 focus:ring-primary',
        className
      )}
      {...props}
    />
  );
}

// Usage: Override padding
<Input className="px-6 py-4" />  // Works correctly!`}
</CodeBlock>
